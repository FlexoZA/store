## products

```sql
create table public.products (
  id bigint generated by default as identity not null,
  created_at timestamp with time zone not null default now(),
  product_name text not null,
  sku text null,
  description text null,
  price numeric null,
  quantity integer null,
  low_stock integer null,
  enabled boolean not null default true,
  material_id bigint null,
  category_id bigint not null default '2'::bigint,
  is_bundled boolean not null default false,
  is_featured boolean not null default false,
  allow_no_stock_sale boolean not null default false,
  constraint Items_pkey primary key (id),
  constraint Items_id_key unique (id),
  constraint products_category_id_fkey foreign KEY (category_id) references categories (id)
) TABLESPACE pg_default;
```

## product_image

```sql
create table public.product_image (
  id bigint generated by default as identity not null,
  created_at timestamp with time zone not null default now(),
  name text null,
  product_id bigint null,
  url text null,
  constraint item_image_pkey primary key (id),
  constraint product_image_product_id_fkey foreign KEY (product_id) references products (id) on update CASCADE on delete CASCADE
) TABLESPACE pg_default;
```

## bundle_products

## profiles

```sql
create table public.profiles (
  id uuid not null,
  name text null,
  surname text null,
  phone_number text null,
  address text null,
  created_at timestamp with time zone not null default timezone ('utc'::text, now()),
  updated_at timestamp with time zone not null default timezone ('utc'::text, now()),
  constraint profiles_pkey primary key (id),
  constraint profiles_id_fkey foreign KEY (id) references auth.users (id) on delete CASCADE
) TABLESPACE pg_default;
```
